## 现有机制梳理

### 1. 金币系统（双货币）

**燃油币（fuelCoins）🪙**

**装备币（gearCoins）⚙️**
- 获取：答复杂题 +8，连击奖励，每日目标

GameState.learning 中已记录：
- `totalWordsSeen` - 见过的单词总数
- `totalWordsMastered` - 掌握的单词总数
- `totalQuizzes` - 完成的答题套数
- `totalQuestions` - 总答题数
- `totalCorrect` - 答对数

### 3. 升级系统（已有）

- engine/tire/body 三种改装
- 等级 1-4 级
- 使用装备币升级

## 增量设计方案

### 核心思路

**不新建复杂系统，在现有数据上直接加一层成就追踪**：

1. 扩展 GameState，增加 `achievements` 字段
2. 在 ProgressTracker 答题时检查成就
3. 成就解锁新赛道

### 3. 成就检查系统

```javascript
// systems/achievement-manager.js

import { ACHIEVEMENTS } from '../config/achievements.js';
import { Events } from '../core/event-bus.js';

export class AchievementManager {
  #eventBus;
  #gameState;

  constructor(eventBus, gameState) {
    this.#eventBus = eventBus;
    this.#gameState = gameState;
  }

  /**
   * 检查所有成就（答题后调用）
   */
  checkAll() {
    const state = this.#gameState.getAll();
    const unlocked = state.achievements || [];

    for (const [id, achievement] of Object.entries(ACHIEVEMENTS)) {
      // 跳过已解锁
      if (unlocked.includes(id)) continue;

      // 检查条件
      if (achievement.check(state)) {
        this.#unlock(achievement);
      }
    }
  }

  /**
   * 解锁成就
   */
  #unlock(achievement) {
    const state = this.#gameState.getAll();

    // 记录成就
    if (!state.achievements.includes(achievement.id)) {
      state.achievements.push(achievement.id);
      this.#gameState.set('achievements', state.achievements);
    }

    // 发放奖励
    if (achievement.reward.track) {
      this.#unlockTrack(achievement.reward.track);
    }
    if (achievement.reward.fuelCoins) {
      this.#gameState.modify('fuelCoins', achievement.reward.fuelCoins);
    }

    // 发送事件
    this.#eventBus.emit(Events.ACHIEVEMENT_UNLOCKED, {
      achievement: achievement
    });

    console.log(`🏆 成就解锁：${achievement.name}`);
  }

  /**
   * 解锁赛道
   */
  #unlockTrack(trackId) {
    const state = this.#gameState.getAll();
    if (!state.unlockedTracks.includes(trackId)) {
      state.unlockedTracks.push(trackId);
      this.#gameState.set('unlockedTracks', state.unlockedTracks);
      console.log(`🏁 赛道解锁：${trackId}`);
    }
  }

  /**
   * 获取所有成就状态
   */
  getAllStatus() {
    const state = this.#gameState.getAll();
    const unlocked = state.achievements || [];

    return Object.values(ACHIEVEMENTS).map(ach => ({
      ...ach,
      unlocked: unlocked.includes(ach.id),
      progress: this.#getProgress(ach, state)
    }));
  }

  /**
   * 获取成就进度（用于 UI 显示）
   */
  #getProgress(achievement, state) {
    // 简单实现：返回当前值和目标值
    // 例如：'quiz-master-10' 返回 { current: 5, target: 10 }
    const match = achievement.description.match(/(\d+)/);
    if (match) {
      const target = parseInt(match[1]);
      if (achievement.id.includes('quiz')) {
        return { current: state.learning.totalQuizzes, target };
      }
      if (achievement.id.includes('word')) {
        return { current: state.learning.totalWordsMastered, target };
      }
    }
    return null;
  }
}
```

### 5. 赛道选择逻辑

```javascript
// js/game.js (改动)

import { TRACK_REGISTRY } from '../config/track-registry.js';

export class Game {
  constructor(canvas) {
    // ... 现有代码 ...

    // 加载上次选择的赛道
    this.selectedTrackId = this.#gameState.get('selectedTrackId') || 'shanghai-2d';
  }

  /**
   * 选择赛道（带检查）
   */
  selectTrack(trackId) {
    const track = TRACK_REGISTRY[trackId];
    if (!track) throw new Error('Unknown track');

    // 检查是否解锁
    const unlocked = this.#gameState.get('unlockedTracks') || [];
    if (!unlocked.includes(trackId)) {
      throw new Error('Track not unlocked');
    }

    // 检查金币
    const fuelCoins = this.#gameState.get('fuelCoins');
    if (fuelCoins < track.cost) {
      throw new Error('Insufficient fuel coins');
    }

    // 保存选择
    this.selectedTrackId = trackId;
    this.#gameState.set('selectedTrackId', trackId);
  }

  /**
   * 开始比赛（扣金币）
   */
  startRace() {
    const track = TRACK_REGISTRY[this.selectedTrackId];

    // 扣除金币
    this.#gameState.modify('fuelCoins', -track.cost);

    // 加载赛道（现有逻辑）
    if (track.type === '2d') {
      this.track = new Track(track.waypoints, track.trackWidth);
    } else {
      // TODO: 3D 赛道
    }

    // 更新渲染系统
    this._renderSystem.setTrack(this.track);

    // 开始倒计时
    this._startCountdown();
  }

  /**
   * 获取可用赛道列表
   */
  getAvailableTracks() {
    const unlocked = this.#gameState.get('unlockedTracks') || [];
    const fuelCoins = this.#gameState.get('fuelCoins');

    return Object.values(TRACK_REGISTRY).map(track => ({
      ...track,
      unlocked: unlocked.includes(track.id),
      canAfford: fuelCoins >= track.cost
    }));
  }
}
```
