## 四、出题策略

### 4.1 每套题构成（10题）

```javascript
/**
 * 出题逻辑：
 * 1. 如果有错词，优先复习（最多3题）
 * 2. 如果有"简单题通过"的词，出复杂题检查
 * 3. 如果有"复杂题通过"的词，出简单题检查
 * 4. 剩余位置出新词
 */

function buildQuiz(words, progress, wrongWords) {
  const quiz = [];
  
  // 1. 错词复习（最多3题）
  const reviewWords = wrongWords.slice(0, 3);
  for (const word of reviewWords) {
    const q = createQuestion(word, getRandomMode(), progress[word.word]);
    q.isReview = true;
    quiz.push(q);
  }
  
  // 2. 待检查词（简单题通过 → 出复杂题检查）
  const needComplexCheck = words.filter(w => 
    progress[w.word]?.status === 'simple_passed' &&
    !progress[w.word]?.complexCorrect
  );
  for (const word of shuffle(needComplexCheck).slice(0, 2)) {
    quiz.push(createQuestion(word, 'RADIO_MSG'));
  }
  
  // 3. 待检查词（复杂题通过 → 出简单题检查）
  const needSimpleCheck = words.filter(w => 
    progress[w.word]?.status === 'complex_passed' &&
    !progress[w.word]?.simpleCorrect
  );
  for (const word of shuffle(needSimpleCheck).slice(0, 2)) {
    quiz.push(createQuestion(word, 'PIT_BOARD'));
  }
  
  // 4. 新词填充到10题
  const newWords = words.filter(w => 
    !progress[w.word] || progress[w.word].status === 'unlearned'
  );
  while (quiz.length < 10) {
    const word = newWords.pop();
    if (word) {
      // 新词先出简单题
      quiz.push(createQuestion(word, 'PIT_BOARD'));
    }
  }
  
  return shuffle(quiz);
}
```

### 4.2 题型分配原则

```
新词：
  - 首次出现：PIT_BOARD（词→中文义）或 STRATEGY（中文义→词）
  - 简单题答对后：下次遇到出 RADIO_MSG（填空）

学习中词汇：
  - 简单题通过 → 出复杂题
  - 复杂题通过 → 出简单题
  - 都通过 → 标记为已掌握，不再出题

错词：
  - 随机出题（优先没答错过的题型）
```

```javascript
const rewardConfig = {

  perCorrectComplex: { fuel: 0, gear: 8 },   // 复杂题答对

  // 连击奖励
  comboBonus: {
    3: { fuel: 0, gear: 5 },   // 连续答对3题
    5: { fuel: 0, gear: 10 },  // 连续答对5题
    7: { fuel: 0, gear: 15 },  // 连续答对7题
    10: { fuel: 0, gear: 25 }, // 连续答对10题（满连）
  },
  
  // 每日目标奖励
  dailyGoals: {
    allThree: { fuel: 50, gear: 30 },        // 完成3套题
    accuracy80: { fuel: 30, gear: 0 },       // 正确率≥80%
    newWords10: { fuel: 20, gear: 0 },       // 新学10个词
  },
};

// 计算示例
function calculateRewards(quizResults) {
  let fuel = 0, gear = 0;
  
  // 每题奖励
  for (const q of quizResults) {
    if (q.correct) {
      if (isSimpleMode(q.mode)) {
        fuel += rewardConfig.perCorrectSimple.fuel;
      } else {
        gear += rewardConfig.perCorrectComplex.gear;
      }
    }
  }
  
  // 连击奖励
  const maxCombo = calculateMaxCombo(quizResults);
  if (rewardConfig.comboBonus[maxCombo]) {
    gear += rewardConfig.comboBonus[maxCombo].gear;
  }
  
  return { fuel, gear };
}
```

### 6.3 自动划分算法（建议）

```javascript
/**
 * 根据以下因素自动计算难度：
 * 1. 词频（语料库频率）
 * 2. 词长
 * 3. 音节复杂度
 * 4. 是否多义词
 */
function calculateLevel(word) {
  let score = 0;
  
  // 词频得分（频率越高越简单）
  const freq = getWordFrequency(word.word);
  if (freq > 10000) score += 1;        // 高频词
  else if (freq > 3000) score += 2;    // 中频词
  else if (freq > 1000) score += 3;    // 较低频
  else score += 4;                     // 低频词
  
  // 词长得分
  if (word.word.length <= 4) score += 0;
  else if (word.word.length <= 6) score += 0.5;
  else if (word.word.length <= 8) score += 1;
  else score += 1.5;
  
  // 音节得分
  const syllables = countSyllables(word.word);
  if (syllables <= 2) score += 0;
  else if (syllables <= 3) score += 0.5;
  else score += 1;
  
  // 多义性
  if (word.meaning_cn.includes('；') || word.meaning_cn.includes('、')) {
    score += 0.5;
  }
  
  // 映射到 1-5
  return Math.min(5, Math.max(1, Math.round(score)));
}
```

## 十、风险与应对

| 风险 | 应对措施 |
|------|----------|
| 孩子快速点完3套题去玩赛车 | 正确率影响金币收入，乱点赚不到金币 |
| 词库难度不合理 | Phase 5 重新划分，持续优化 |
| 孩子只做简单题 | 简单题只给燃油币，装备币（买Nitro/升级）需要复杂题 |
| 中途退出丢失进度 | `wr_quiz_session` 保存进度，重新打开恢复 |
| 数据量太大（2000词进度） | 只存储已出现过的词，压缩存储 |
