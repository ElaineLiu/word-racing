## 设计目标

支持两套独立的赛车系统并行开发：
1. **2D赛道系统**：基于现有架构，开发多条不同风格的赛道
2. **3D赛道系统**：基于Three.js的全新实现

两套系统共享学习进度和金币系统，通过配置开关切换，独立开发、独立部署。

```javascript
// systems/track-factory.js

export class TrackFactory {
  static create(trackId, config = {}) {
    const trackData = TRACK_REGISTRY[trackId];
    if (!trackData) throw new Error(`Unknown track: ${trackId}`);

    // 根据 feature flag 选择实现
    const use3D = FeatureFlags.isEnabled('3d-track');

    if (use3D && trackData.type === '3d') {
      return new Track3D(trackData, config);
    } else {
      return new Track2D(trackData, config);
    }
  }

  static getAvailableTracks(gameState) {
    return Object.values(TRACK_REGISTRY).filter(track =>
      TrackUnlockManager.isUnlocked(track.id, gameState)
    );
  }
}
```

### 4. 赛道解锁系统

```javascript
// systems/track-unlock-manager.js

export class TrackUnlockManager {
  /**
   * 检查赛道是否解锁
   * @param {string} trackId - 赛道ID
   * @param {GameState} gameState - 游戏状态（含学习进度）
   */
  static isUnlocked(trackId, gameState) {
    const track = TRACK_REGISTRY[trackId];
    if (!track) return false;

    const { unlockRequirements } = track;
    const progress = gameState.getLearningProgress();

    // 检查各项条件
    if (unlockRequirements.wordsLearned &&
        progress.wordsLearned < unlockRequirements.wordsLearned) {
      return false;
    }

    if (unlockRequirements.masteryCount &&
        progress.masteryCount < unlockRequirements.masteryCount) {
      return false;
    }

    if (unlockRequirements.quizzesCompleted &&
        progress.quizzesCompleted < unlockRequirements.quizzesCompleted) {
      return false;
    }

    return true;
  }

  /**
   * 获取解锁进度
   */
  static getUnlockProgress(trackId, gameState) {
    const track = TRACK_REGISTRY[trackId];
    if (!track) return null;

    const progress = gameState.getLearningProgress();
    const requirements = track.unlockRequirements;

    return {
      wordsLearned: {
        current: progress.wordsLearned,
        required: requirements.wordsLearned || 0
      },
      masteryCount: {
        current: progress.masteryCount,
        required: requirements.masteryCount || 0
      },
      quizzesCompleted: {
        current: progress.quizzesCompleted,
        required: requirements.quizzesCompleted || 0
      }
    };
  }
}
```

### 5. 金币消耗系统

```javascript
// systems/racing-cost-manager.js

export class RacingCostManager {
  /**
   * 检查是否有足够金币
   */
  static canAfford(trackId, gameState) {
    const track = TRACK_REGISTRY[trackId];
    if (!track) return false;

    return gameState.fuelCoins >= track.cost;
  }

  /**
   * 扣除金币并开始比赛
   */
  static startRace(trackId, gameState) {
    const track = TRACK_REGISTRY[trackId];
    if (!track) throw new Error('Unknown track');

    if (!this.canAfford(trackId, gameState)) {
      throw new Error('Insufficient fuel coins');
    }

    // 扣除金币
    gameState.fuelCoins -= track.cost;
    gameState.save();

    return true;
  }

  /**
   * 获取退款（比赛取消时）
   */
  static refund(trackId, gameState) {
    const track = TRACK_REGISTRY[trackId];
    if (track) {
      gameState.fuelCoins += track.cost;
      gameState.save();
    }
  }
}
```

### 6. Feature Flag 系统

```javascript
// config/feature-flags.js

export const FeatureFlags = {
  flags: {
    '2d-track': true,      // 2D赛道总开关
    '3d-track': false,     // 3D赛道总开关（开发完成后开启）
    'multiple-tracks': true // 多赛道功能
  },

  isEnabled(flagName) {
    return this.flags[flagName] === true;
  },

  enable(flagName) {
    this.flags[flagName] = true;
  },

  disable(flagName) {
    this.flags[flagName] = false;
  },

  // 从 localStorage 加载用户配置
  load() {
    try {
      const saved = localStorage.getItem('wr_feature_flags');
      if (saved) {
        this.flags = { ...this.flags, ...JSON.parse(saved) };
      }
    } catch (e) {
      console.error('Failed to load feature flags:', e);
    }
  },

  // 保存到 localStorage
  save() {
    localStorage.setItem('wr_feature_flags', JSON.stringify(this.flags));
  }
};
```

**Why**: 运行时动态控制功能开关，无需重新部署。

**How to apply**: 在初始化时加载配置，根据flag决定启用哪些赛道。

### 7. Game 类改造

```javascript
// js/game.js (改造后)

import { TrackFactory } from '../systems/track-factory.js';
import { TrackUnlockManager } from '../systems/track-unlock-manager.js';
import { RacingCostManager } from '../systems/racing-cost-manager.js';
import { FeatureFlags } from '../config/feature-flags.js';

export class Game {
  constructor(canvas, options = {}) {
    // ... 现有代码 ...

    // 赛道选择
    this.selectedTrackId = options.trackId || 'shanghai-2d';
    this.track = null;  // 延迟初始化

    // 加载 feature flags
    FeatureFlags.load();
  }

  /**
   * 选择赛道
   */
  selectTrack(trackId) {
    const track = TRACK_REGISTRY[trackId];
    if (!track) throw new Error('Unknown track');

    // 检查是否解锁
    if (!TrackUnlockManager.isUnlocked(trackId, this.#gameState)) {
      throw new Error('Track not unlocked');
    }

    // 检查金币
    if (!RacingCostManager.canAfford(trackId, this.#gameState)) {
      throw new Error('Insufficient fuel coins');
    }

    this.selectedTrackId = trackId;
  }

  /**
   * 开始比赛
   */
  startRace() {
    // 扣除金币
    RacingCostManager.startRace(this.selectedTrackId, this.#gameState);

    // 创建赛道实例
    this.track = TrackFactory.create(this.selectedTrackId);

    // 更新渲染系统
    this._renderSystem.setTrack(this.track);

    // 开始比赛逻辑
    this._startCountdown();
  }

  /**
   * 获取可用赛道列表
   */
  getAvailableTracks() {
    return TrackFactory.getAvailableTracks(this.#gameState);
  }
}
```

### 8. UI 改造：赛道选择界面

```javascript
// views/track-selection-view.js

export class TrackSelectionView {
  constructor(game, container) {
    this.game = game;
    this.container = container;
  }

  render() {
    const tracks = this.game.getAvailableTracks();

    this.container.innerHTML = `
      <div class="track-selection">
        <h2>选择赛道</h2>
        <div class="track-grid">
          ${tracks.map(track => this._renderTrackCard(track)).join('')}
        </div>
      </div>
    `;

    this._bindEvents();
  }

  _renderTrackCard(track) {
    const isUnlocked = TrackUnlockManager.isUnlocked(track.id, this.game.#gameState);
    const canAfford = RacingCostManager.canAfford(track.id, this.game.#gameState);
    const progress = TrackUnlockManager.getUnlockProgress(track.id, this.game.#gameState);

    return `
      <div class="track-card ${isUnlocked ? 'unlocked' : 'locked'}">
        <img src="${track.thumbnail}" alt="${track.name}">
        <h3>${track.name}</h3>
        <p>${track.description}</p>

        ${isUnlocked ? `
          <div class="cost">
            🪙 ${track.cost} 燃油币
            ${canAfford ? '' : '<span class="insufficient">（金币不足）</span>'}
          </div>
        ` : `
          <div class="unlock-requirements">
            <div>已学单词：${progress.wordsLearned.current}/${progress.wordsLearned.required}</div>
            <div>已掌握：${progress.masteryCount.current}/${progress.masteryCount.required}</div>
          </div>
        `}

        <button
          class="select-btn"
          data-track-id="${track.id}"
          ${!isUnlocked || !canAfford ? 'disabled' : ''}>
          ${isUnlocked ? '选择' : '未解锁'}
        </button>
      </div>
    `;
  }

  _bindEvents() {
    this.container.querySelectorAll('.select-btn').forEach(btn => {
      btn.addEventListener('click', () => {
        const trackId = btn.dataset.trackId;
        this.game.selectTrack(trackId);
        this.game.startRace();
      });
    });
  }
}
```

## 开发分支策略

### 分支规划

```
main (当前)
  │
  ├── feature/track-system-foundation  (基础架构)
  │     - 抽象接口定义
  │     - 工厂模式
  │     - 解锁系统
  │     - 金币消耗
  │     - Feature flags
  │
  ├── feature/2d-tracks (2D多赛道)
  │     - Track2D 实现
  │     - 3条新赛道数据
  │     - 2D赛道选择UI
  │
  └── feature/3d-track (3D赛道)
        - Track3D 实现（Three.js）
        - 3D渲染系统
        - 3D物理引擎
        - 3D赛道场景
```

### 开发顺序

1. **Phase 0**: 在 main 分支开发基础架构（抽象接口、解锁系统、金币消耗）
2. **Phase 1**: 从 main 切出 `feature/2d-tracks`，开发2D多赛道
3. **Phase 2**: 从 main 切出 `feature/3d-track`，开发3D赛道
4. **Phase 3**: 两个分支分别测试完成后，合并回 main

**Why**: 先建基础架构，再并行开发，避免后期重构。

**How to apply**: 按顺序执行，每个阶段完成后合并。

## 风险与对策

### 风险1：3D性能问题
**对策**:
- 添加性能检测，自动降级到2D
- 提供画质设置选项
- 3D赛道设置更高的解锁门槛（确保用户设备足够好）

### 风险2：金币经济失衡
**对策**:
- 赛道消耗与金币获取速率匹配
- 添加每日首胜奖励（额外金币）
- 考虑添加观看广告获取金币的选项

### 风险3：两套系统维护成本
**对策**:
- 共享尽可能多的代码（物理、UI、状态管理）
- 统一测试框架
- 清晰的接口文档
